const API_URL = window.location.hostname === "localhost" || window.location.hostname === "127.0.0.1" ? "http://127.0.0.1:5000/api" : window.location.origin + "/api"; let config = JSON.parse(localStorage.getItem("config_gestor_alunos")); let alunosAtuais = []; function iniciar() { if (config) { document.getElementById('login-area').style.display = 'none'; document.getElementById('unidade-nome').innerText = config.nome; carregarAlunos(); carregarHistorico(); } } function switchTab(tabId, element) { document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active')); document.querySelectorAll('.nav-link').forEach(link => link.classList.remove('active')); document.getElementById(tabId).classList.add('active'); element.classList.add('active'); } function limparFiltroTextoData() { if(document.getElementById('add-data-texto')) { document.getElementById('add-data-texto').value = ""; } document.getElementById('filtro-data-texto').value = ""; } function limparFiltroCalendario() { document.getElementById('filtro-data').value = ""; } async function fazerLogin() { const cnpjInput = document.getElementById('l-cnpj').value; const cnpj = cnpjInput.replace(/\D/g, ''); const senha = document.getElementById('l-senha').value.trim(); if (!cnpj || !senha) { alert("Por favor, preencha o CNPJ e a Senha."); return; } try { const res = await fetch(`${API_URL}/clientes/login-tablet`, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ cnpj, senha }) }); if (res.ok) { const dados = await res.json(); localStorage.setItem("config_gestor_alunos", JSON.stringify({ id: dados.id, nome: dados.nome })); location.reload(); } else { alert("Acesso negado! Verifique suas credenciais."); } } catch (e) { alert("Erro de comunicação com o servidor."); } } function filtrarHistorico() { const dataCalendario = document.getElementById('filtro-data').value; const dataTexto = document.getElementById('filtro-data-texto').value.trim(); const salaSel = document.getElementById('filtro-sala').value.toLowerCase().trim(); const linhas = document.querySelectorAll('#lista-historico-completo tr'); linhas.forEach(linha => { const dataTxt = linha.children[0].innerText.trim(); const salaTxt = linha.children[4].innerText.toLowerCase(); let bateData = false; if (!dataCalendario && !dataTexto) { bateData = true; } else if (dataCalendario) { bateData = (dataTxt === dataCalendario); } else if (dataTexto) { bateData = dataTxt.includes(dataTexto); } const bateSala = !salaSel || salaTxt.includes(salaSel); if (bateData && bateSala) { linha.style.display = ""; } else { linha.style.display = "none"; } }); } async function carregarAlunos() { try { const res = await fetch(`${API_URL}/alunos/unidade/${config.id}`); alunosAtuais = await res.json(); document.getElementById('dash-total-alunos').innerText = alunosAtuais.length; document.getElementById('lista-alunos').innerHTML = alunosAtuais.map(a => ` ${a.nome}
